    Define JourneyNodeWidget extends StatefulWidget with constructor:
      const JourneyNodeWidget({
        super.key,
        required this.glyph,
        required this.name,
        required this.state,
        required this.onTap,
      });
      final String glyph;
      final String name;
      final JourneyNodeState state;
      final VoidCallback? onTap;  // null for future/locked nodes (already inert)

    Define _JourneyNodeWidgetState extends State<JourneyNodeWidget>
    with SingleTickerProviderStateMixin:

    ANIMATION (pulse glow on current node only, D-14):
      late AnimationController _pulseController;
      late Animation<double> _pulseSpread; // spread radius lerps 0→14

      In initState():
        _pulseController = AnimationController(
          vsync: this,
          duration: const Duration(milliseconds: 2400),
        );
        _pulseSpread = Tween<double>(begin: 0.0, end: 14.0).animate(
          CurvedAnimation(parent: _pulseController, curve: QalamMotion.easeInOut),
        );
        if (widget.state == JourneyNodeState.current) {
          _pulseController.repeat(reverse: true);
        }

      In dispose(): _pulseController.dispose();

      In didUpdateWidget(): if state changes to/from current, start/stop the animation.

    BUILD METHOD:
    Wrap the entire column in a GestureDetector:
      GestureDetector(
        onTap: widget.onTap,  // null for future/locked → GestureDetector ignores null
        child: Column(mainAxisSize: MainAxisSize.min, children: [
          Stack(alignment: Alignment.topCenter, clipBehavior: Clip.none, children: [
            // TODAY pill (current node only)
            if (widget.state == JourneyNodeState.current)
              Positioned(top: -28, child: _TodayChip()),
            // Animated node circle
            AnimatedBuilder(
              animation: _pulseSpread,
              builder: (context, child) => _buildCircle(child!),
              child: ArabicText(widget.glyph, style: const TextStyle(fontSize: 32)),
            ),
            // Gold star badge (complete nodes only — D-13, REWARDS ONLY)
            if (widget.state == JourneyNodeState.complete)
              Positioned(top: -4, right: -5, child: _StarBadge()),
          ]),
          const SizedBox(height: 7),
          Text(
            widget.name,
            style: QalamTextStyles.label.copyWith(
              color: (widget.state == JourneyNodeState.future || widget.state == JourneyNodeState.locked)
                ? QalamColors.fgMuted
                : QalamColors.fg,
            ),
            textAlign: TextAlign.center,
          ),
        ]),
      )

    _buildCircle(Widget child) returns a Container(
      width: 68, height: 68,
      decoration: BoxDecoration(
        shape: BoxShape.circle,
        color: _circleColor(),
        boxShadow: _buildShadow(),
      ),
      alignment: Alignment.center,
      child: child,
    )
    For future nodes, wrap in a Stack with a CustomPaint overlay for the dashed border
    (reuse _DashedCirclePainter from plan 03.1-02, or extract it into a shared file if
    journey_screen.dart already defines it). If _DashedCirclePainter is private to
    journey_screen.dart, move it to a new shared file
    lib/features/journey/widgets/dashed_circle_painter.dart and import from both.

    _circleColor() returns:
      JourneyNodeState.complete → QalamColors.success
      JourneyNodeState.current  → QalamColors.primary
      JourneyNodeState.future   → QalamColors.surfaceRaised (white)
      JourneyNodeState.locked   → const Color(0xFFCDD8DA)

    _buildShadow() returns the animated BoxShadow list:
      complete: [BoxShadow(color: const Color(0xFF2A8A60), offset: Offset(0, 5))]
      current: [
        BoxShadow(color: QalamColors.primaryPressed, offset: Offset(0, 5)),
        BoxShadow(
          color: QalamColors.primary.withOpacity(0.45),
          blurRadius: _pulseSpread.value,
          spreadRadius: _pulseSpread.value,
        ),
      ]
      future/locked: []

    _StarBadge is a private Row with 3 star icons:
      Row(mainAxisSize: MainAxisSize.min, children: [
        _Star(), _Star(), _Star(),
      ])
    _Star() returns a CustomPaint(size: Size(11, 11), painter: _StarPainter(color: QalamColors.reward))
    Reuse the existing _StarPainter logic from mastery_celebration.dart (the 5-pointed star path
    math is already tested and correct — copy the class into this file as a private painter).
    QalamColors.reward (gold) is used ONLY for these mastery-badge stars (REWARDS ONLY rule,
    D-13). NO other use of QalamColors.reward on the Journey screen.

    ArabicText inside the circle: use ArabicText(glyph) which already handles RTL correctly.
    Text color: complete/current → Colors.white; future/locked → QalamColors.fgMuted.
    Pass via textStyle: ArabicText(glyph, style: TextStyle(fontSize: 32, color: textColor)).

    IMPORTS:
      import 'dart:math' as math;

      import '../../../theme/colors.dart';

      import '../../../theme/text_styles.dart';
      import '../../../widgets/arabic_text.dart';
